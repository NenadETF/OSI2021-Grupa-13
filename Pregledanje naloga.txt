#include <stdio.h>
#include <stdlib.h>
#include <string.h>
typedef struct { char ime[20], prezime[20], korisnicko_ime[30], sifra[7];} KORISNIK;

void pregled_korisnika_iz_datoteke()
{

    FILE *dat;
    KORISNIK k;
    if(dat=fopen("korisnicki_nalozi.dat", "rb")!=NULL)
    {
      while(fread(&k, sizeof(KORISNIK),1, dat))
        {
            printf("Ime %s Prezime %s Korisnicko ime %s, sifra %s\n", k.ime, k.prezime, k.korisnicko_ime, k.sifra);
        }
       fclose(dat);
    }
    else ("Greska kod otvaranja\n");

}
